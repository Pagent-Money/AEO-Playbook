IndexNow Integration - Immediate Content Indexing

IndexNow enables instant URL submission to search engines when content changes. Based on IndexNow.org documentation, this protocol notifies search engines immediately about content updates.

Implementation Strategy

Purpose: Accelerate search engine discovery and AI crawler indexing

Supported Engines: Bing, Seznam, Yandex, and other participating search engines

Benefit: Reduces crawl delay from days/weeks to minutes/hours

Step 1: Generate and Host IndexNow Key

// Generate IndexNow key (8-128 hexadecimal characters)
function generateIndexNowKey(length = 64) {
  const chars = '0123456789abcdefABCDEF';
  let key = '';
  for (let i = 0; i < length; i++) {
    key += chars[Math.floor(Math.random() * chars.length)];
  }
  return key;
}

// Example key: a1b2c3d4e5f6789012345678901234567890abcdef1234567890

Step 2: Single URL Submission

// Submit single URL when content changes
async function submitToIndexNow(url, key) {
  const endpoints = [
    'https://api.indexnow.org/indexnow',
    'https://www.bing.com/indexnow',
    'https://search.seznam.cz/indexnow'
  ];
  
  const results = [];
  
  for (const endpoint of endpoints) {
    try {
      const response = await fetch(
        `${endpoint}?url=${encodeURIComponent(url)}&key=${key}`,
        { 
          method: 'GET',
          headers: {
            'User-Agent': 'AEO-Indexer/1.0'
          }
        }
      );
      
      const result = {
        endpoint: endpoint,
        status: response.status,
        success: response.status === 200 || response.status === 202
      };
      
      if (response.status === 200) {
        console.log(`‚úÖ ${endpoint}: URL submitted successfully`);
      } else if (response.status === 202) {
        console.log(`‚è≥ ${endpoint}: URL received, validation pending`);
      } else if (response.status === 400) {
        console.log(`‚ùå ${endpoint}: Bad request - check URL format`);
      } else if (response.status === 403) {
        console.log(`‚ùå ${endpoint}: Forbidden - key validation failed`);
      } else if (response.status === 429) {
        console.log(`‚ö†Ô∏è ${endpoint}: Too many requests - retry later`);
      }
      
      results.push(result);
      
    } catch (error) {
      console.error(`‚ùå ${endpoint}: Network error`, error);
      results.push({ endpoint, error: error.message, success: false });
    }
  }
  
  return results;
}

// Usage: Submit when content changes
const key = 'your-indexnow-key-here';
submitToIndexNow('https://yourdomain.com/new-article.html', key);

Step 3: Batch URL Submission (Up to 10,000 URLs)

// Submit multiple URLs in a single request
async function submitBatchToIndexNow(urls, key, host) {
  const payload = {
    host: host,
    key: key,
    urlList: urls.slice(0, 10000) // Max 10,000 URLs per request
  };
  
  const endpoints = [
    'https://api.indexnow.org/indexnow',
    'https://www.bing.com/indexnow'
  ];
  
  const results = [];
  
  for (const endpoint of endpoints) {
    try {
      const response = await fetch(endpoint, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json; charset=utf-8',
          'User-Agent': 'AEO-Indexer/1.0'
        },
        body: JSON.stringify(payload)
      });
      
      const result = {
        endpoint: endpoint,
        status: response.status,
        urlCount: urls.length,
        success: response.status === 200 || response.status === 202
      };
      
      if (response.status === 200) {
        console.log(`‚úÖ ${endpoint}: ${urls.length} URLs submitted successfully`);
      } else {
        console.log(`‚ùå ${endpoint}: Batch submission failed (${response.status})`);
      }
      
      results.push(result);
      
    } catch (error) {
      console.error(`‚ùå ${endpoint}: Batch submission error`, error);
      results.push({ endpoint, error: error.message, success: false });
    }
  }
  
  return results;
}

// Usage: Batch submit multiple pages
const updatedPages = [
  'https://yourdomain.com/page1.html',
  'https://yourdomain.com/page2.html',
  'https://yourdomain.com/changelog.html'
];

submitBatchToIndexNow(updatedPages, 'your-key', 'yourdomain.com');

Step 4: Automated CMS Integration

// IndexNow automation class for content management systems
class IndexNowManager {
  constructor(key, host, options = {}) {
    this.key = key;
    this.host = host;
    this.queue = new Set();
    this.options = {
      batchDelay: options.batchDelay || 5000,    // 5 second delay
      maxBatchSize: options.maxBatchSize || 100,  // Max URLs per batch
      retryDelay: options.retryDelay || 30000,    // 30 second retry delay
      maxRetries: options.maxRetries || 3
    };
    this.retryCount = new Map();
  }
  
  // Add URL to submission queue
  queueURL(url) {
    this.queue.add(url);
    this.scheduleSubmission();
  }
  
  // Schedule batch submission with delay
  scheduleSubmission() {
    clearTimeout(this.submitTimer);
    this.submitTimer = setTimeout(() => {
      this.processBatch();
    }, this.options.batchDelay);
  }
  
  // Process queued URLs
  async processBatch() {
    if (this.queue.size === 0) return;
    
    const urls = Array.from(this.queue).slice(0, this.options.maxBatchSize);
    this.queue.clear();
    
    try {
      if (urls.length === 1) {
        await submitToIndexNow(urls[0], this.key);
      } else {
        await submitBatchToIndexNow(urls, this.key, this.host);
      }
      
      // Clear retry counts for successful submissions
      urls.forEach(url => this.retryCount.delete(url));
      
    } catch (error) {
      console.error('IndexNow submission failed:', error);
      this.handleRetries(urls);
    }
  }
  
  // Handle failed submissions with retry logic
  handleRetries(urls) {
    urls.forEach(url => {
      const retries = this.retryCount.get(url) || 0;
      if (retries < this.options.maxRetries) {
        this.retryCount.set(url, retries + 1);
        setTimeout(() => {
          this.queueURL(url);
        }, this.options.retryDelay * (retries + 1));
      } else {
        console.error(`Max retries exceeded for URL: ${url}`);
        this.retryCount.delete(url);
      }
    });
  }
  
  // Submit changelog updates (important for AEO)
  submitChangelogUpdate(changelogURL) {
    console.log('üìù Submitting changelog update to IndexNow...');
    this.queueURL(changelogURL);
  }
  
  // Submit new content immediately
  submitImmediately(url) {
    console.log(`üöÄ Immediate IndexNow submission: ${url}`);
    return submitToIndexNow(url, this.key);
  }
}

// Usage in your CMS or static site generator
const indexNowManager = new IndexNowManager('your-key', 'yourdomain.com');

// Hook into content publishing workflow
function onContentPublished(url) {
  indexNowManager.queueURL(url);
}

function onContentUpdated(url) {
  indexNowManager.queueURL(url);
}

function onChangelogUpdated() {
  indexNowManager.submitChangelogUpdate('https://yourdomain.com/changelog.html');
}
