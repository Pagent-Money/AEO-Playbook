// AI Platform Detection
function detectAIPlatform(referrer) {
  const aiPlatforms = {
    'chat.openai.com': 'ChatGPT',
    'perplexity.ai': 'Perplexity',
    'copilot.microsoft.com': 'Copilot',
    'claude.ai': 'Claude',
    'chat.deepseek.com': 'DeepSeek'
  };
  
  for (const [domain, platform] of Object.entries(aiPlatforms)) {
    if (referrer.includes(domain)) {
      return platform;
    }
  }
  return 'Other';
}

// Track AI Citations
function trackAICitation(platform, citationType) {
  gtag('event', 'ai_citation', {
    'ai_platform': platform,
    'citation_type': citationType,
    'timestamp': new Date().toISOString()
  });
}

Validation and Testing Templates

Schema Validation Script

Quality Assurance Protocol

Validation Frequency: Weekly automated checks

Tools: Google Rich Results Test, Schema.org Validator

Performance Threshold: 100% schema validation required

#!/bin/bash
# AEO Schema Validation Script

# Define URLs to validate
URLS=(
  "https://yourdomain.com/"
  "https://yourdomain.com/documentation"
  "https://yourdomain.com/faq"
)

# Google Rich Results Test API
RICH_RESULTS_API="https://searchconsole.googleapis.com/v1/urlTestingTools/richResults:run"

# Validate each URL
for url in "${URLS[@]}"; do
  echo "Validating: $url"
  
  # Check schema markup
  curl -X POST "$RICH_RESULTS_API" \
    -H "Content-Type: application/json" \
    -d "{\"url\": \"$url\"}" \
    --silent | jq '.richResultsResult'
  
  # Basic accessibility check
  lighthouse --only-categories=accessibility "$url" --output=json
  
  echo "---"
done

# Performance validation
echo "Running Core Web Vitals check..."
lighthouse --only-categories=performance "$URLS[0]" --output=json
